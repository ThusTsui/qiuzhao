## 栈和队列

[232.用栈实现队列](https://leetcode.cn/problems/implement-queue-using-stacks/description/)

```C++
class MyQueue {
public:
    stack<int> sin;
    stack<int> sout;
    MyQueue() {

    }
    
    void push(int x) {
        sin.push(x);
    }
    
    int pop() {
        if(sout.empty()){
            while(!sin.empty()){
                sout.push(sin.top());
                sin.pop();
            }            
        }
        int result = sout.top();
        sout.pop();
        return result;
    }
    
    int peek() {
        int res = this->pop();
        sout.push(res);
        return res;
    }
    
    bool empty() {
        return sin.empty() && sout.empty();
    }
};
```


[225. 用队列实现栈](https://leetcode.cn/problems/implement-stack-using-queues/)

- 只用一个队列实现的关键在于pop时，将除队尾元素之外的值再重新插一遍队尾并删除
```C++
class MyStack {
public:
    queue<int> qu;
    MyStack() {

    }
    
    void push(int x) {
        qu.push(x);
    }
    
    int pop() {
        int size = qu.size();
        size--;
        while(size--){
            qu.push(qu.front());
            qu.pop();
        }
        int res = qu.front();
        qu.pop();
        return res;
    }
    
    int top() {
        return qu.back();
    }
    
    bool empty() {
        return qu.empty();
    }
};

```
